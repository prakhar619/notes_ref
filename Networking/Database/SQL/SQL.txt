



Foreign Keys
    CREATE TABLE parent (
        id INT NOT NULL,
            PRIMARY KEY (id)
    ) ENGINE=INNODB;

    CREATE TABLE child (
        id INT,
        parent_id INT,
            INDEX par_ind (parent_id),
            FOREIGN KEY (parent_id)
                REFERENCES parent(id)
                ON UPDATE RESTRICT          (or CASCADE)
                ON DELETE RESTRICT          (or CASCADE)
    )ENGINE=INNODB;
